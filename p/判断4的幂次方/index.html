public:
    bool isPowerOfFour(int n) {
        if (n <= 0) return false;
        int score = 0;
        while (n > 1) {
            if (n % 2 != 0) return false;
            score++;
            n /= 2;
        }
        return (score % 2 == 0);
    }
};
